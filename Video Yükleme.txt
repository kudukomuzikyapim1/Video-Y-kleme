        // Dosya DoÄŸrulama Fonksiyonu
        function validateFile(file, allowedTypes, maxSize) {
            const fileExtension = file.name.split('.').pop().toLowerCase();
            const isValidType = allowedTypes.includes(fileExtension);
            const isValidSize = file.size <= maxSize;
            
            if (!isValidType) {
                throw new Error(`Desteklenmeyen dosya formatÄ±: ${fileExtension}. Desteklenen formatlar: ${allowedTypes.join(', ')}`);
            }
            
            if (!isValidSize) {
                throw new Error(`Dosya boyutu Ã§ok bÃ¼yÃ¼k: ${formatFileSize(file.size)}. Maksimum boyut: ${formatFileSize(maxSize)}`);
            }
            
            return true;
        }

        // Dosya boyutunu formatlama
        function formatFileSize(bytes) {
            if (bytes === 0) return '0 Bytes';
            const k = 1024;
            const sizes = ['Bytes', 'KB', 'MB', 'GB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
        }

            // GerÃ§ek YouTube API yÃ¼kleme fonksiyonu
            async function performYouTubeUpload(title, description, privacyStatus, file, format) {
                return new Promise(async (resolve, reject) => {
                    try {
                        showLoading('YouTube\'a YÃ¼kleniyor', 'Video hazÄ±rlanÄ±yor...');
                        
                        const youtubeUploadUrl = 'https://www.googleapis.com/upload/youtube/v3/videos';
                        
                        const metadata = {
                            snippet: {
                                title: title,
                                description: description,
                                tags: ['mÃ¼zik', 'audio', 'kudukomÃ¼zikyapÄ±m'],
                                categoryId: '10'
                            },
                            status: {
                                privacyStatus: privacyStatus,
                                selfDeclaredMadeForKids: false
                            }
                        };

                        let contentType = 'video/mp4';
                        switch(format) {
                            case 'mp3':
                                contentType = 'audio/mpeg';
                                break;
                            case 'mp4':
                                contentType = 'video/mp4';
                                break;
                            case 'm4a':
                                contentType = 'audio/mp4';
                                break;
                            case 'wav':
                                contentType = 'audio/wav';
                                break;
                            case 'flac':
                                contentType = 'audio/flac';
                                break;
                        }

                        document.getElementById('statusFileStatus').textContent = 'Upload baÅŸlatÄ±lÄ±yor...';
                        
                        const initiateResponse = await fetch(`${youtubeUploadUrl}?uploadType=resumable&part=snippet,status`, {
                            method: 'POST',
                            headers: {
                                'Authorization': `Bearer ${accessToken}`,
                                'Content-Type': 'application/json',
                                'X-Upload-Content-Length': file.size,
                                'X-Upload-Content-Type': contentType
                            },
                            body: JSON.stringify(metadata)
                        });

                        if (!initiateResponse.ok) {
                            throw new Error('Upload baÅŸlatÄ±lamadÄ±: ' + initiateResponse.statusText);
                        }

                        const uploadUrl = initiateResponse.headers.get('Location');
                        if (!uploadUrl) {
                            throw new Error('Upload URL alÄ±namadÄ±');
                        }

                        const CHUNK_SIZE = 256 * 1024;
                        const fileSize = file.size;
                        let uploadedBytes = 0;

                        const updateProgress = (bytesUploaded) => {
                            const progress = Math.round((bytesUploaded / fileSize) * 100);
                            document.getElementById('progressFill').style.width = progress + '%';
                            document.getElementById('statusProgress').textContent = progress + '%';
                            document.getElementById('statusFileStatus').textContent = `${progress}% yÃ¼klendi`;
                            document.getElementById('progressText').textContent = 
                                `YÃ¼kleniyor: ${Math.round(uploadedBytes / 1024 / 1024)}MB / ${Math.round(fileSize / 1024 / 1024)}MB`;
                        };

                        document.getElementById('statusFileStatus').textContent = 'Dosya yÃ¼kleniyor...';

                        for (let start = 0; start < fileSize; start += CHUNK_SIZE) {
                            const end = Math.min(start + CHUNK_SIZE, fileSize);
                            const chunk = file.slice(start, end);
                            
                            const chunkResponse = await fetch(uploadUrl, {
                                method: 'PUT',
                                headers: {
                                    'Content-Range': `bytes ${start}-${end - 1}/${fileSize}`,
                                    'Content-Type': contentType,
                                    'Authorization': `Bearer ${accessToken}`
                                },
                                body: chunk
                            });

                            if (!chunkResponse.ok && chunkResponse.status !== 308) {
                                throw new Error('Chunk yÃ¼kleme hatasÄ±: ' + chunkResponse.statusText);
                            }

                            uploadedBytes = end;
                            updateProgress(uploadedBytes);
                        }

                        document.getElementById('statusFileStatus').textContent = 'Upload tamamlanÄ±yor...';

                        const finalResponse = await fetch(uploadUrl, {
                            method: 'PUT',
                            headers: {
                                'Content-Range': `bytes */${fileSize}`,
                                'Authorization': `Bearer ${accessToken}`
                            }
                        });

                        if (!finalResponse.ok) {
                            throw new Error('Upload tamamlanamadÄ±: ' + finalResponse.statusText);
                        }

                        const result = await finalResponse.json();
                        
                        if (!result.id) {
                            throw new Error('Video ID alÄ±namadÄ±');
                        }

                        const thumbnailFile = document.getElementById('imageFile').files[0];
                        if (thumbnailFile) {
                            await uploadThumbnail(result.id, thumbnailFile);
                        }

                        hideLoading();
                        resolve(result.id);

                    } catch (error) {
                        hideLoading();
                        reject(error);
                    }
                });
            }
